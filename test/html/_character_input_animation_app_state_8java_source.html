96  private void assignJoysticks(){
97  Joystick[] joysticks = inputManager.getJoysticks();
98  if (joysticks != null){
99  for( Joystick j : joysticks ) {
100  for(JoystickAxis axis : j.getAxes()){
101  if(axis == j.getXAxis()){
102  axis.assignAxis(InputMapping.StrafeRight.name(), InputMapping.StrafeLeft.name());
103  } else if ( axis == j.getYAxis()){
104  axis.assignAxis(InputMapping.MoveBackward.name(), InputMapping.MoveForward.name());
105  } else if (axis.getLogicalId().equals("ry")){
106  axis.assignAxis(ChaseCamera.ChaseCamUp, ChaseCamera.ChaseCamDown);
107  } else if(axis.getLogicalId().equals("rx")){
108  axis.assignAxis(ChaseCamera.ChaseCamMoveRight, ChaseCamera.ChaseCamMoveLeft);
109  }
110  }
111 
112  for(JoystickButton button : j.getButtons()){
113  button.assignButton("Fire");
114  }
